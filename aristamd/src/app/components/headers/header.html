Options
  • isExpanded: boolean
  • titleLeft: string
  • contentLeft: string (template is acceptable)
  • titleRight: string
  • contentRight: string (template is acceptable)
  • nudge: boolean
  • transitionTime: number (300 ms default)
  • app-header-directive : directive (required)
